判断规则
def classify(paper):
    # 规则 A: 渠道新增
    has_channel = any(kw in paper.abstract.lower()
        for kw in ['new channel', 'add platform', 'launch'])

    # 规则 B: 实证分析
    is_empirical = paper.method in ['survey', 'experiment', 'secondary data']

    # 排除非学术类型
    if paper.type == 'review' or paper.type == 'theoretical':
        return 'N', '非实证研究'

    if has_channel and is_empirical:
        return 'Y', '匹配渠道新增 + 实证分析'
    elif has_channel or is_empirical:
        return '?', '部分匹配，建议全文确认'
    else:
        return 'N', '不涉及渠道新增'
